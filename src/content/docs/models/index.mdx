import { db, PFU_HHKB_Models, Pcbs, Controllers } from 'astro:db';
import { eq } from 'astro:db';

export const models = await db.select().from(PFU_HHKB_Models);

export const generations = models.reduce((acc, model) => {
  if (!acc[model.generation]) {
    acc[model.generation] = [];
  }
  acc[model.generation].push(model);
  return acc;
}, {});

# PFU HHKB Models

Official HHKB keyboards manufactured by PFU, organized by generation.

{Object.entries(generations).map(async ([generation, genModels]) => {
  return (
    <div key={generation} style="margin: 2rem 0;">
      <h2>{generation}</h2>
      <div style="display: grid; gap: 1rem;">
        {genModels.map(async (model) => {
          const pcb = await db.select().from(Pcbs).where(eq(Pcbs.id, model.pcb_id)).get();
          const controller = await db.select().from(Controllers).where(eq(Controllers.id, model.controller_id)).get();

          return (
            <div key={model.id} style="border: 1px solid #ccc; padding: 1rem; border-radius: 8px;">
              <h3>{model.model_name} {model.type_s ? 'Type-S' : ''}</h3>
              <div style="display: grid; grid-template-columns: repeat(2, 1fr); gap: 0.5rem; font-size: 0.9rem;">
                <div><strong>Model Number:</strong> {model.model_number}</div>
                <div><strong>Layout:</strong> {model.layout}</div>
                <div><strong>Case Color:</strong> {model.case_color}</div>
                <div><strong>Keycap Color:</strong> {model.keycap_color}</div>
                <div><strong>Legends:</strong> {model.legends}</div>
                <div><strong>Type-S:</strong> {model.type_s ? 'Yes' : 'No'}</div>
                {model.release_date && (
                  <div><strong>Release Date:</strong> {new Date(model.release_date).toLocaleDateString()}</div>
                )}
              </div>

              {(pcb || controller) && (
                <div style="margin-top: 1rem; padding-top: 1rem; border-top: 1px solid #eee;">
                  <h4 style="margin: 0 0 0.5rem 0; font-size: 0.9rem;">Components</h4>
                  <div style="font-size: 0.85rem;">
                    {pcb && <div>• PCB: {pcb.name}</div>}
                    {controller && <div>• Controller: {controller.name}</div>}
                  </div>
                </div>
              )}

              {model.notes && (
                <div style="margin-top: 0.5rem; font-size: 0.85rem; font-style: italic; color: #666;">
                  {model.notes}
                </div>
              )}
            </div>
          );
        })}
      </div>
    </div>
  );
})}
